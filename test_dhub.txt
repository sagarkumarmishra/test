```python
from notebookutils import mssparkutils
import json
from datetime import datetime
import requests

print("=== Starting DataHub Upload ===")

# Read token from Pipeline (fixed name)
token = mssparkutils.notebook.getParameter("bearer_token")

print("Token received (first 30 chars):", token[:30] + "..." if token else "EMPTY!")

    raise Exception("Token is missing or too short! Check Pipeline parameter name.")

# Prepare your data
data = {
    "eventTime": datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ"),
    "policyId": "POL123",
    "status": "ACTIVE",
    "amount": 1000
}

json_content = json.dumps(data, ensure_ascii=False)

print("Data prepared:", json_content)

# Upload to DataHub
url = "https://dhub-int.cta.allianz/api/v1/file/upload"

headers = {
    "Authorization": f"Bearer {token}",
    "Accept": "application/json",
    "tenant": "AZRE"
}

files = {
    "file": ("test.json", json_content, "application/json")
}

print("Sending multipart upload...")

response = requests.post(url, headers=headers, files=files, timeout=60)

print("Status Code:", response.status_code)
print("Response:", response.text)

if response.status_code == 200:
    print("✅ SUCCESS: File uploaded to DataHub!")
else:
    print("❌ FAILED")
    raise Exception(f"Upload failed with status {response.status_code}: {response.text}")
```
