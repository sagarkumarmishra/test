```python
from notebookutils import mssparkutils   # Keep this import for safety
import json
from datetime import datetime
import requests

print("=== Starting DataHub Upload ===")

# === READ TOKEN FROM PIPELINE (Most Reliable Way) ===
token = mssparkutils.notebook.getContext().get("bearer_token")

print("Token received (first 40 chars):", 
      token[:40] + "..." if token and len(token) > 40 else "EMPTY or MISSING!")

if not token or len(token) < 50:
    raise Exception("❌ Token is missing or too short! "
                    "Make sure parameter name in Pipeline is exactly 'bearer_token' (case-sensitive)")

# === PREPARE YOUR DATA ===
data = {
    "eventTime": datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%SZ"),
    "policyId": "POL123",
    "status": "ACTIVE",
    "amount": 1000
}

json_content = json.dumps(data)

print("Data prepared for upload:", json_content)

# === SEND MULTIPART UPLOAD TO DATAHUB ===
url = "https://dhub-int.cta.allianz/api/v1/file/upload"

headers = {
    "Authorization": f"Bearer {token}",
    "Accept": "application/json",
    "tenant": "AZRE"
}

files = {
    "file": ("test.json", json_content, "application/json")
}

print("Sending multipart POST to DataHub...")

response = requests.post(url, headers=headers, files=files, timeout=60)

print("Status Code:", response.status_code)
print("Response Text:", response.text[:500])  # limit output if too long

if response.status_code == 200:
    print("✅ SUCCESS: Data uploaded to DataHub!")
else:
    print("❌ FAILED")
    raise Exception(f"Upload failed - Status: {response.status_code} | Details: {response.text}")
```
